\label{def:ff-fourier-set}
\lean{BlrPcp.HasFourierSet}

For $f : \mathbb{F}_q^n \to \mathbb{F}_q$, define its Fourier set
  \[
    \Phi(f) := \{\varphi_c\}_{c \in \mathbb{F}_q^\times},
    \qquad
    \varphi_c(x) := \omega_p^{\mathrm{Tr}(c f(x))}.
  \]
\end{definition}

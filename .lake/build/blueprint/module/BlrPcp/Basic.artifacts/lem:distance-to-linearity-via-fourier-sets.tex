  \[
    \Delta(f,\mathrm{LIN})
    = 1 - \frac{1}{q}\left(1 + \max_{\alpha \in \mathbb{F}_q^n}
      \sum_{c \in \mathbb{F}_q^\times} \widehat{\varphi_c}(c\alpha)\right).
  \]

\begin{definition}
\label{def:finite-field-blr-test}
\lean{BlrPcp.HasFiniteFieldBLRTest}

\leanok
The finite-field BLR test samples $x,y \in F^n$ and
  $a,b \in F \setminus \{0\}$, and checks whether
  \[
    a f(x) + b f(y) = f(ax + by).
  \]
\end{definition}

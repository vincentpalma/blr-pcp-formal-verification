\begin{definition}
\label{def:ff-additive-characters}
\lean{BlrPcp.HasAdditiveCharacters}

\leanok
Write $q = p^e$. For each $\alpha \in \mathbb{F}_q^n$, define
  the additive character
  \[
    \chi_\alpha(x) := \omega_p^{\mathrm{Tr}(\langle \alpha, x\rangle)},
  \]
  where $\mathrm{Tr} : \mathbb{F}_q \to \mathbb{F}_p$ is the field trace.
\end{definition}
